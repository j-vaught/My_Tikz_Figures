  \def\hity{0}
  \def\gothit{0}

  % Check circle
  \path[name intersections={of=beam and circ, total=\t}]
    \pgfextra{
      \ifnum\t>0
        \foreach \i in {1,...,\t} {
          \pgfpointanchor{intersection-\i}{center}
          \pgfgetlastxy{\ix}{\iy}
          \pgfmathparse{veclen(\ix,\iy)}
          \let\d\pgfmathresult
          \pgfmathparse{\d < \mindist ? 1 : 0}
          \ifdim\pgfmathresult pt>0.5pt
            \global\edef\mindist{\d}
            \global\edef\hitx{\ix}
            \global\edef\hity{\iy}
            \global\def\gothit{1}
          \fi
        }
